# 学习助手 - 数据库设计

## 1. 数据库概述

### 1.1 数据库选型
- **数据库**: PostgreSQL 15+
- **选型理由**:
  - 支持 JSONB，适合存储灵活的数据结构
  - 强大的全文搜索能力
  - 优秀的并发性能
  - 丰富的数据类型和扩展
  - 支持窗口函数和复杂查询

### 1.2 设计规范
- 所有表使用 `id` 作为主键（UUID 类型）
- 所有表包含 `created_at` 和 `updated_at` 字段
- 使用 `soft_delete` 策略（可选）
- 外键使用 `ON DELETE CASCADE/SET NULL`
- 适当使用 JSONB 存储灵活数据

### 1.3 命名规范
- 表名：复数形式，snake_case（如：`users`, `questions`）
- 字段名：snake_case
- 索引名：`idx_{表名}_{字段名}`
- 唯一约束：`uk_{表名}_{字段名}`

## 2. 数据表设计

### 2.1 用户相关表

#### 2.1.1 用户表 (users)

```sql
-- 用户表
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    phone VARCHAR(20) NOT NULL UNIQUE,
    password_hash VARCHAR(255),  -- 密码登录时存储
    nickname VARCHAR(50) NOT NULL,
    avatar_url VARCHAR(500),     -- 头像 URL
    school VARCHAR(100),         -- 学校
    major VARCHAR(100),          -- 专业
    grade VARCHAR(20),           -- 年级
    
    -- 学习偏好
    preferred_subjects TEXT[],   -- 偏好学科
    learning_style VARCHAR(20),  -- 学习风格
    
    -- 状态
    status VARCHAR(20) DEFAULT 'active',  -- active, banned, deleted
    last_login_at TIMESTAMPTZ,
    last_login_ip INET,
    
    -- 元数据
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    deleted_at TIMESTAMPTZ
);

-- 索引
CREATE INDEX idx_users_phone ON users(phone);
CREATE INDEX idx_users_status ON users(status);
CREATE INDEX idx_users_created_at ON users(created_at);

-- 软删除视图
CREATE VIEW users_active AS
SELECT * FROM users WHERE deleted_at IS NULL;
```

### 2.2 题目相关表

#### 2.2.1 题目表 (questions)

```sql
-- 题目表
CREATE TABLE questions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    
    -- 题目内容
    original_text TEXT,              -- 识别出的原始文本
    question_type VARCHAR(20),       -- 选择题，填空题，解答题，判断题
    subject VARCHAR(50),             -- 学科：数学，物理，化学，英语
    grade_level VARCHAR(20),         -- 难度：基础，进阶，提高
    
    -- 图片
    images JSONB NOT NULL,           -- [{url, type, width, height}]
    
    -- 解析内容
    solution TEXT,                   -- 解题步骤
    answer TEXT,                     -- 答案
    knowledge_points JSONB,          -- 知识点列表 [{name, level, description}]
    
    -- AI 生成内容
    similar_questions JSONB,         -- 生成的类似题目 [{question, answer, difficulty}]
    
    -- 追问信息
    follow_up_count INTEGER DEFAULT 0,  -- 追问次数
    max_follow_ups INTEGER DEFAULT 10,  -- 最大追问次数
    
    -- 状态
    status VARCHAR(20) DEFAULT 'completed',  -- processing, completed, failed
    error_message TEXT,
    
    -- 元数据
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    completed_at TIMESTAMPTZ,
    
    -- 软删除
    deleted_at TIMESTAMPTZ
);

-- 索引
CREATE INDEX idx_questions_user_id ON questions(user_id);
CREATE INDEX idx_questions_subject ON questions(subject);
CREATE INDEX idx_questions_status ON questions(status);
CREATE INDEX idx_questions_created_at ON questions(created_at);
CREATE INDEX idx_questions_knowledge_points ON questions USING gin(knowledge_points);

-- 注释
COMMENT ON TABLE questions IS '题目表，存储用户上传的题目及解析';
```

### 3.3 创建视图和函数

```sql
-- 创建用户活跃视图
CREATE OR REPLACE VIEW users_active AS
SELECT * FROM users WHERE deleted_at IS NULL;

-- 创建题目统计视图
CREATE OR REPLACE VIEW question_stats_by_user AS
SELECT 
    user_id,
    COUNT(*) as total_questions,
    COUNT(*) FILTER (WHERE status = 'completed') as completed_questions,
    COUNT(*) FILTER (WHERE follow_up_count > 0) as questions_with_followups,
    AVG(follow_up_count) as avg_followups
FROM questions
WHERE deleted_at IS NULL
GROUP BY user_id;

-- 创建错题统计视图
CREATE OR REPLACE VIEW mistake_stats_by_user AS
SELECT 
    user_id,
    COUNT(*) as total_mistakes,
    COUNT(*) FILTER (WHERE status = 'new') as new_mistakes,
    COUNT(*) FILTER (WHERE status = 'mastered') as mastered_mistakes,
    COUNT(*) FILTER (WHERE status = 'reviewing') as reviewing_mistakes
FROM mistakes
WHERE deleted_at IS NULL
GROUP BY user_id;

-- 创建更新 updated_at 的触发器函数
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 为所有需要 updated_at 的表添加触发器
CREATE TRIGGER update_users_updated_at
    BEFORE UPDATE ON users
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_questions_updated_at
    BEFORE UPDATE ON questions
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_coursewares_updated_at
    BEFORE UPDATE ON coursewares
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_mistakes_updated_at
    BEFORE UPDATE ON mistakes
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();
```

## 6. 数据备份策略

### 6.1 备份脚本

```bash
#!/bin/bash
# backup.sh

DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR="/backups/study_helper"
DB_NAME="study_helper"
DB_USER="postgres"

# 全量备份
pg_dump -U $DB_USER -Fc $DB_NAME > $BACKUP_DIR/full_$DATE.dump

# 只备份结构
pg_dump -U $DB_USER -s $DB_NAME > $BACKUP_DIR/schema_$DATE.sql

# 只备份数据
pg_dump -U $DB_USER -a $DB_NAME > $BACKUP_DIR/data_$DATE.sql

# 删除 7 天前的备份
find $BACKUP_DIR -name "*.dump" -mtime +7 -delete
find $BACKUP_DIR -name "*.sql" -mtime +7 -delete
```

```bash
# 恢复数据库
pg_restore -U postgres -d study_helper -c backup_20260317_120000.dump
```
