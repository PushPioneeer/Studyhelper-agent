# 学习助手 - 前端设计文档

## 1. 技术栈选型

### 1.1 核心技术栈

根据 **vercel-react-native-skills** 的最佳实践，选择以下技术栈：

```
React Native (Expo)
├── TypeScript           # 类型安全
├── Zustand              # 状态管理（轻量、简单）
├── React Navigation     # 路由导航
├── FlashList            # 高性能列表
├── expo-image           # 图片加载优化
├── Reanimated           # 动画
├── Gesture Handler      # 手势
└── Axios                # HTTP 请求
```

### 1.2 选型理由

**状态管理 - Zustand**
- ✅ 轻量级（<1KB）
- ✅ API 简单，学习成本低
- ✅ 支持 TypeScript
- ✅ 无需 Provider 包裹
- ✅ 符合 react-state-minimize 规则

**列表组件 - FlashList**
- ✅ 比 FlatList 快 3-5 倍
- ✅ 自动内存优化
- ✅ 符合 list-performance-virtualize 规则
- ✅ Shopify 开源，维护活跃

**图片组件 - expo-image**
- ✅ 支持 WebP 格式
- ✅ 自动缓存
- ✅ 支持渐进式加载
- ✅ 符合 ui-expo-image 规则

**导航 - React Navigation**
- ✅ Native Stack 性能最优
- ✅ 符合 navigation-native-navigators 规则
- ✅ 类型安全
- ✅ 社区活跃

### 3.1 用户状态

```typescript
// store/useAuthStore.ts
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface User {
  id: string;
  phone: string;
  nickname: string;
  avatar_url?: string;
  school?: string;
  major?: string;
  grade?: string;
}

interface AuthState {
  // 数据
  user: User | null;
  token: string | null;
  refreshToken: string | null;
  isAuthenticated: boolean;
  
  // Actions
  login: (token: string, refreshToken: string, user: User) => Promise<void>;
  logout: () => void;
  updateUser: (user: Partial<User>) => void;
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set) => ({
      user: null,
      token: null,
      refreshToken: null,
      isAuthenticated: false,
      
      login: async (token, refreshToken, user) => {
        set({
          token,
          refreshToken,
          user,
          isAuthenticated: true,
        });
      },
      
      logout: () => {
        set({
          user: null,
          token: null,
          refreshToken: null,
          isAuthenticated: false,
        });
      },
      
      updateUser: (userData) => {
        set((state) => ({
          user: state.user ? { ...state.user, ...userData } : null,
        }));
      },
    }),
    {
      name: 'auth-storage',
    }
  )
);
```

```typescript
// store/useQuestionStore.ts
import { create } from 'zustand';

interface Question {
  id: string;
  original_text: string;
  question_type: string;
  subject: string;
  images: Array<{ url: string; width: number; height: number }>;
  solution: string;
  answer: string;
  knowledge_points: Array<{ name: string; level: string; description: string }>;
  follow_up_count: number;
  max_follow_ups: number;
  created_at: string;
}

interface FollowUp {
  id: string;
  message: string;
  response: string;
  created_at: string;
}

interface QuestionState {
  // 数据
  currentQuestion: Question | null;
  followUps: FollowUp[];
  questions: Question[];
  
  // Actions
  setCurrentQuestion: (question: Question) => void;
  addFollowUp: (followUp: FollowUp) => void;
  setQuestions: (questions: Question[]) => void;
  clearCurrentQuestion: () => void;
}

export const useQuestionStore = create<QuestionState>((set) => ({
  currentQuestion: null,
  followUps: [],
  questions: [],
  
  setCurrentQuestion: (question) => {
    set({ currentQuestion: question, followUps: [] });
  },
  
  addFollowUp: (followUp) => {
    set((state) => ({
      followUps: [...state.followUps, followUp],
      currentQuestion: state.currentQuestion
        ? {
            ...state.currentQuestion,
            follow_up_count: state.currentQuestion.follow_up_count + 1,
          }
        : null,
    }));
  },
  
  setQuestions: (questions) => {
    set({ questions });
  },
  
  clearCurrentQuestion: () => {
    set({ currentQuestion: null, followUps: [] });
  },
}));
```

```typescript
// store/useMistakeStore.ts
import { create } from 'zustand';

interface Mistake {
  id: string;
  original_text: string;
  subject: string;
  thumbnail?: string;
  correct_answer: string;
  user_answer: string;
  error_type: string;
  status: 'new' | 'reviewing' | 'mastered' | 'archived';
  mastery_level: number;
  tags: string[];
  created_at: string;
}

interface MistakeState {
  // 数据
  mistakes: Mistake[];
  selectedIds: string[];
  
  // Actions
  setMistakes: (mistakes: Mistake[]) => void;
  toggleSelect: (id: string) => void;
  clearSelection: () => void;
  addMistake: (mistake: Mistake) => void;
  updateMistake: (id: string, updates: Partial<Mistake>) => void;
}

export const useMistakeStore = create<MistakeState>((set) => ({
  mistakes: [],
  selectedIds: [],
  
  setMistakes: (mistakes) => {
    set({ mistakes });
  },
  
  toggleSelect: (id) => {
    set((state) => ({
      selectedIds: state.selectedIds.includes(id)
        ? state.selectedIds.filter((sid) => sid !== id)
        : [...state.selectedIds, id],
    }));
  },
  
  clearSelection: () => {
    set({ selectedIds: [] });
  },
  
  addMistake: (mistake) => {
    set((state) => ({
      mistakes: [mistake, ...state.mistakes],
    }));
  },
  
  updateMistake: (id, updates) => {
    set((state) => ({
      mistakes: state.mistakes.map((m) =>
        m.id === id ? { ...m, ...updates } : m
      ),
    }));
  },
}));
```

```typescript
import React, { useState, useCallback } from 'react';
import { View, StyleSheet, TouchableOpacity } from 'react-native';
import { Camera } from 'expo-camera';
import { Button } from '../../components/common/Button';
import { useQuestionStore } from '../../store/useQuestionStore';

export const CameraScreen: React.FC = () => {
  const [hasPermission, setHasPermission] = useState<boolean | null>(null);

  const [takingPicture, setTakingPicture] = useState(false);

    if (!camera || takingPicture) return;
    
    setTakingPicture(true);
    try {
      const photo = await camera.takePictureAsync({
        quality: 0.8,
      });
      
      // 上传图片并获取解析结果

      setTakingPicture(false);

    return <View />;
  }
  
  if (hasPermission === false) {
    return <Text>需要相机权限</Text>;

  return (
    <View style={styles.container}>
      <Camera
        style={styles.camera}
        type={Camera.Constants.Type.back}
        ref={(ref) => setCamera(ref)}
      />
      
      <View style={styles.buttonContainer}>

          title="拍照解题"
          onPress={takePicture}
          loading={takingPicture}

};
```

## 7. API 服务层

### 7.1 Axios 配置

```typescript
// services/api.ts
import axios from 'axios';
import { useAuthStore } from '../store/useAuthStore';

const API_BASE_URL = 'http://localhost:3000/api/v1';

export const api = axios.create({
  baseURL: API_BASE_URL,
  timeout: 30000,
  headers: {
    'Content-Type': 'application/json',
  },
});

// 请求拦截器
api.interceptors.request.use(
  (config) => {
    const token = useAuthStore.getState().token;
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  (error) => Promise.reject(error)
);

// 响应拦截器
api.interceptors.response.use(
  (response) => response.data,
  async (error) => {
    const originalRequest = error.config;

    // Token 过期，尝试刷新
    if (error.response?.status === 401 && !originalRequest._retry) {
      originalRequest._retry = true;
      
      try {
        const refreshToken = useAuthStore.getState().refreshToken;
        const response = await axios.post(`${API_BASE_URL}/auth/refresh`, {
          refresh_token: refreshToken,
        });
        
        useAuthStore.getState().login(
          response.data.token,
          response.data.refresh_token,
          useAuthStore.getState().user
        );
        
        originalRequest.headers.Authorization = `Bearer ${response.data.token}`;
        return api(originalRequest);
      } catch (refreshError) {
        useAuthStore.getState().logout();
        return Promise.reject(refreshError);
      }
    }
    
    return Promise.reject(error);
  }
);
```
